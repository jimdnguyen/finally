```typescript
'use client'

import { useState } from 'react'
import { usePriceStore } from '@/store/priceStore'
import { usePortfolio } from '@/hooks/usePortfolio'
import { WatchlistPanel } from '@/components/watchlist/WatchlistPanel'
import { MainChart } from '@/components/charts/MainChart'
import { Treemap } from '@/components/charts/Treemap'
import { PnLChart } from '@/components/charts/PnLChart'
import { PositionsTable } from '@/components/charts/PositionsTable'
import { TradeBar } from '@/components/header/TradeBar'
import { ConnectionStatus } from '@/components/header/ConnectionStatus'

export default function Page() {
  const [selectedTicker, setSelectedTicker] = useState<string | null>(null)
  const status = usePriceStore((s) => s.status)
  const { data: portfolio } = usePortfolio()

  // Default to first available price if nothing selected
  const prices = usePriceStore((s) => s.prices)
  const defaultTicker =
    selectedTicker || Object.keys(prices)[0] || 'AAPL'
  const displayTicker = selectedTicker || defaultTicker

  return (
    <main className="flex flex-col h-full bg-base">
      {/* Header: Portfolio value, connection status, trade bar */}
      <header className="px-6 py-4 border-b border-gray-700 flex items-center justify-between">
        <div className="flex items-center gap-4">
          <h1 className="text-2xl font-bold text-accent-yellow">
            FinAlly
          </h1>
          <div className="flex flex-col">
            <span className="text-3xl font-semibold text-gray-100">
              ${portfolio?.total_value.toFixed(2) || '10,000.00'}
            </span>
            <span className="text-xs text-gray-400">
              Cash: ${portfolio?.cash_balance.toFixed(2) || '10,000.00'}
            </span>
          </div>
        </div>

        <div className="flex items-center gap-6">
          <div className="flex items-center gap-2">
            <ConnectionStatus status={status} />
            <span className="text-xs text-gray-400">
              {status === 'live'
                ? 'Connected'
                : status === 'reconnecting'
                  ? 'Reconnecting...'
                  : 'Connecting...'}
            </span>
          </div>

          <TradeBar selectedTicker={displayTicker} />
        </div>
      </header>

      {/* Main content grid */}
      <div className="flex-1 overflow-hidden grid grid-cols-3 gap-6 p-6">
        {/* Left: Main chart (large) */}
        <div className="col-span-2 flex flex-col gap-6">
          {/* Chart */}
          <div className="flex-1 bg-panel rounded border border-gray-700 p-4">
            <h2 className="text-sm font-semibold text-gray-100 mb-4">
              {displayTicker} Price Action
            </h2>
            <MainChart ticker={displayTicker} />
          </div>

          {/* Portfolio charts row */}
          <div className="grid grid-cols-2 gap-6">
            {/* Treemap */}
            <div className="bg-panel rounded border border-gray-700 p-4">
              <h2 className="text-sm font-semibold text-gray-100 mb-4">
                Portfolio Allocation
              </h2>
              <Treemap />
            </div>

            {/* P&L Chart */}
            <div className="bg-panel rounded border border-gray-700 p-4">
              <h2 className="text-sm font-semibold text-gray-100 mb-4">
                P&L History
              </h2>
              <PnLChart />
            </div>
          </div>
        </div>

        {/* Right: Positions table */}
        <div className="flex flex-col bg-panel rounded border border-gray-700 p-4">
          <h2 className="text-sm font-semibold text-gray-100 mb-4">
            Positions
          </h2>
          <PositionsTable />
        </div>
      </div>
    </main>
  )
}
```

Key design decisions:
1. **State management**: selectedTicker via useState (local), portfolio via usePortfolio hook (global), prices via usePriceStore selector (Zustand)
2. **Layout**: Header spans full width with portfolio total and trade bar; below is 3-column content (2:1 ratio)
3. **Chart selection**: Clicking a ticker in WatchlistPanel (via prop) updates selectedTicker state here
4. **Data flow**: All components consume from Zustand/TanStack Query, no prop drilling beyond selectedTicker
5. **Color scheme**: Uses Tailwind custom colors from tailwind.config.ts (bg-base, bg-panel, text-accent-yellow, etc.)

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| UI ↔ Backend API | All API calls to `/api/*` endpoints; frontend assumes responses are valid and trusted |
| Watchlist clicks → State updates | Clicking a ticker in watchlist updates local selectedTicker state; no external call |
| Chart data → Zustand | Prices from SSE written to Zustand store by usePriceStream hook; store is source of truth |
| Trade bar → POST /api/portfolio/trade | Client-side validation of qty; backend validates cash/shares and executes |
| Chat input → LLM → Auto-execution | Frontend sends user message to backend; LLM generates trade actions; backend validates and executes; frontend displays confirmations |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-08 | Tampering | Watchlist ticker selection | Accept | Local state only; no external persistence; user clicking their own UI |
| T-04-09 | Information Disclosure | Portfolio total value in header | Accept | User's own data, publicly visible to them; no other users |
| T-04-10 | Repudiation | Trade bar execution | Mitigate | All trades logged in backend `trades` table with timestamp and user_id; frontend displays confirmation inline |
| T-04-11 | Denial of Service | Rapid clicking Buy button | Mitigate | Mutation isPending state disables button during request; TanStack Query retry=1 limits retries |
| T-04-12 | Elevation of Privilege | Accessing other user data | N/A | Single-user system; all hardcoded as user_id="default" |

All security enforcement delegated to backend. Frontend is display layer only.
</threat_model>
