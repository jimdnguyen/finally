import pytest

def test_buy_success_with_sufficient_cash(client: TestClient, test_db, price_cache):
    """Buy 10 AAPL with sufficient cash; position created, cash decreases."""
    # Setup: ensure AAPL price in cache
    price_cache.update("AAPL", 190.0)
    
    # Act
    response = client.post(
        "/api/portfolio/trade",
        json={"ticker": "AAPL", "quantity": 10, "side": "buy"}
    )
    
    # Assert
    assert response.status_code == 200
    data = response.json()
    assert data["executed_quantity"] == 10
    assert data["executed_price"] == 190.0
    
    # Verify in DB
    result = test_db.execute(text("SELECT quantity FROM positions WHERE ticker='AAPL'"))
    assert result.scalar() == 10.0

def test_buy_fails_insufficient_cash(client: TestClient, price_cache):
    """Buy 100,000 shares without sufficient cash; returns 400."""
    price_cache.update("AAPL", 190.0)
    
    response = client.post(
        "/api/portfolio/trade",
        json={"ticker": "AAPL", "quantity": 100000, "side": "buy"}
    )
    
    assert response.status_code == 400
    detail = response.json().get("detail", "").lower()
    assert "insufficient" in detail or "cash" in detail

def test_sell_fails_oversell(client: TestClient, price_cache):

    price_cache.update("UNKNOWN", 100.0)
    
    response = client.post(
        "/api/portfolio/trade",
        json={"ticker": "UNKNOWN", "quantity": 50, "side": "sell"}

# Test imports
import pytest
import json
from pydantic import ValidationError
from app.chat.models import ChatResponse, Trade, WatchlistChange

# TEST-02: LLM response parsing and validation

def test_valid_chat_response_parsing():
    """Parse valid structured LLM response with all fields."""
    json_str = '''
    {
        "message": "Here's my analysis of your portfolio...",
        "trades": [
            {"ticker": "AAPL", "side": "buy", "quantity": 10}
        ],
        "watchlist_changes": [
            {"ticker": "PYPL", "action": "add"}
        ]
    }
    '''
    response = ChatResponse.model_validate_json(json_str)
    assert response.message == "Here's my analysis of your portfolio..."
    assert len(response.trades) == 1
    assert response.trades[0].ticker == "AAPL"
    assert response.trades[0].quantity == 10
    assert len(response.watchlist_changes) == 1
    assert response.watchlist_changes[0].action == "add"

def test_valid_chat_response_empty_arrays():
    """Parse response with empty trades and watchlist arrays."""
    json_str = '''
    {
        "message": "No trades for you today.",
        "trades": [],
        "watchlist_changes": []
    }
    '''
    response = ChatResponse.model_validate_json(json_str)
    assert response.message == "No trades for you today."
    assert response.trades == []
    assert response.watchlist_changes == []

def test_valid_chat_response_only_message():
    """Parse response with only message field (trades/watchlist optional)."""
    json_str = '{"message": "Hello, I can help with trading."}'
    response = ChatResponse.model_validate_json(json_str)
    assert response.message == "Hello, I can help with trading."
    # trades and watchlist_changes should be [] or None depending on schema

def test_malformed_json_rejected():
    """Reject malformed JSON (syntax error)."""
    json_str = '{"message": "Hello", "trades": [invalid]}'
    with pytest.raises((ValueError, json.JSONDecodeError)):
        ChatResponse.model_validate_json(json_str)

def test_missing_message_field_rejected():
    """Reject response missing required 'message' field."""
    json_str = '{"trades": [], "watchlist_changes": []}'
    with pytest.raises(ValidationError):
        ChatResponse.model_validate_json(json_str)

def test_invalid_trade_missing_quantity():
    """Reject trade object without quantity field."""
    json_str = '''
    {
        "message": "Trade rejected",
        "trades": [
            {"ticker": "AAPL", "side": "buy"}
        ],
        "watchlist_changes": []
    }
    '''
    with pytest.raises(ValidationError):
        ChatResponse.model_validate_json(json_str)

def test_invalid_trade_bad_side():
    """Reject trade with invalid side (not 'buy' or 'sell')."""

    {
        "message": "Trade rejected",

        "watchlist_changes": []
    }
